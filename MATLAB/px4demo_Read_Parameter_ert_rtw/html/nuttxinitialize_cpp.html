37void setTaskPeriod(double periodInSeconds, int sigNo) {



38    timer_t timerId;



39    struct sigevent sev;



40    struct itimerspec its;



41    long stNanoSec;



42    int status;



43



44    /* Create a timer */



45    sev.sigev_notify = SIGEV_SIGNAL;



46    sev.sigev_signo = sigNo;



47    sev.sigev_value.sival_ptr = &timerId;



48    status = timer_create(CLOCK_REALTIME, &sev, &timerId);



49    CHECK_STATUS(status, 0, "timer_create");



50



51    /* Arm real-time scheduling timer */



52    stNanoSec = (long)(periodInSeconds * 1e9);



53    its.it_value.tv_sec = stNanoSec / 1000000000;



54    its.it_value.tv_nsec = stNanoSec % 1000000000;



55    its.it_interval.tv_sec = its.it_value.tv_sec;



56    its.it_interval.tv_nsec = its.it_value.tv_nsec;



57    status = timer_settime(timerId, 0, &its, NULL);



58    CHECK_STATUS(status, 0, "timer_settime");



59}

123void myAddBlockForThisEvent(int sigNo) {



124    int status;



125    sigset_t sigMask;



126



127    sigemptyset(&sigMask);



128    sigaddset(&sigMask, sigNo);



129    status = pthread_sigmask(SIG_BLOCK, &sigMask, NULL);

133void nuttxRTOSInit(double baseRatePeriod, int numSubrates) {



134    /* This is because in Daren's original Task callback ,a sleep was added before creating the



135     * threads*/



136    sleep(1);



137



138    int status;



139



140#if defined(MW_PX4_NUTTX_BUILD)



141    size_t stackSize;



142    int baseRatePriority = MW_BASERATE_PRIORITY;



143#endif



144



145#ifdef MW_HAS_MULTIPLE_RATES



146    int i;



147    // char taskName[20];



148#endif



149#if defined(MW_PX4_NUTTX_BUILD)



150    sched_lock();



151#endif



152    // status = sem_init(&termSem, 0, 0); PX4 PSP Dev: termSem no longer needed



153    // CHECK_STATUS(status, 0,"sem_init:termSem");



154    status = sem_init(&stopSem, 0, 0);



155    CHECK_STATUS(status, 0, "sem_init:stopSem");



156    status = sem_init(&baserateTaskSem, 0, 0);



157    CHECK_STATUS(status, 0, "sem_init:baserateTaskSem");



158



159    // or you can use/check: _POSIX_PRIORITY_SCHEDULING



160    // _POSIX_THREAD_PRIORITY_SCHEDULING



161#if !defined(_POSIX_PRIORITY_SCHEDULING)



162    PX4_INFO("Priority scheduling is NOT supported by your system.");



163    PX4_INFO("The generated code will not run correctly because your");



164    PX4_INFO("model contains multiple rates and uses multi-tasking");



165    PX4_INFO("code generation mode. You can only run the generated code");



166    PX4_INFO("in single-tasking mode in your system. Open");



167    PX4_INFO("Simulation -> Configuration Parameters -> Solver dialog");



168    PX4_INFO("and set \"Tasking mode for periodic sample times\" parameter to SingleTasking.");



169    PX4_INFO(



170        "Re-build the Simulink model with the new settings and try executing the generated code "



171        "again.");



172    fflush(stdout);



173    exit(-1);



174#endif



175



176#if defined(MW_PX4_NUTTX_BUILD)



177    /* Set scheduling policy of the main thread to SCHED_FIFO */



178    g_sp.sched_priority = sched_get_priority_max(SCHED_FIFO) - 50;



179    status = sched_setscheduler(0, SCHED_FIFO, &g_sp);



180    CHECK_STATUS(status, 0, "sched_setscheduler");



181



182    /*Added init attribute and scheduler policy */



183    pthread_attr_init(&attr);



184    status = pthread_attr_setinheritsched(&attr, PTHREAD_EXPLICIT_SCHED);



185    CHECK_STATUS(status, 0, "pthread_attr_setinheritsched");



186    status = pthread_attr_setschedpolicy(&attr, SCHED_FIFO);



187    CHECK_STATUS(status, 0, "pthread_attr_setschedpolicy");



188    // status = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);



189    // CHECK_STATUS(status, 0, "pthread_attr_setdetachstate");



190



191    /* PTHREAD_STACK_MIN is the minimum stack size required to start a thread */



192    stackSize = 8192 + PTHREAD_STACK_MIN;



193



194    /*added stack attribute back in */



195    status = pthread_attr_setstacksize(&attr, stackSize);



196    CHECK_STATUS(status, 0, "pthread_attr_setstacksize");



197#endif



198    /* set up info structure */



199    g_info.period = baseRatePeriod;



200    g_info.sigNo = BASERATE_TIMER_SIGNAL; // SIGRTMIN;



201



202#if MW_RTOS_DEBUG



203    PX4_INFO("   stackSize = %d sched_priority = %d", stackSize, g_sp.sched_priority);



204    PX4_INFO("   MW_BASERATE_PERIOD = %8.5f MW_BASERATE_PRIORITY = %d SIGRTMIN = 0x%08X",



205             (double)baseRatePeriod, (int)baseRatePriority, SIGRTMIN);



206    PX4_INFO("   Init info.period = %8.5f sigNo = 0x%04X", g_info.period, g_info.sigNo);



207    PX4_INFO("**creating the Base Rate thread before calling pthread_create**");



208    fflush(stdout);



209#endif



210



211    /* Create the Base Rate Task here */



212#if defined(MW_PX4_NUTTX_BUILD)



213    g_sp.sched_priority = baseRatePriority;



214    status = pthread_attr_setschedparam(&attr, &g_sp);



215    status = pthread_create(&baseRateThread, &attr, &baseRateTask, (void*)&g_info);



216#elif defined(MW_PX4_POSIX_BUILD)



217    status = pthread_create(&baseRateThread, nullptr, &baseRateTask, (void*)&g_info);



218#endif



219#if MW_RTOS_DEBUG



220    PX4_INFO("** Base Rate Task ID = %d with Priority = %d\n", baseRateThread, g_sp.sched_priority);



221    fflush(stdout);



222#endif



223



224/* Create sub-rate Tasks here */



225#ifdef MW_HAS_MULTIPLE_RATES



226#if MW_RTOS_DEBUG



227    PX4_INFO("Creating sub-rate task threads\n");



228    PX4_INFO("Number of sub rate tasks: %d \n", MW_NUMBER_SUBRATES);



229    fflush(stdout);



230#endif



231



232    for (i = 0; i < MW_NUMBER_SUBRATES; i++) {



233        taskId[i] = i;



234        status = sem_init(&subrateTaskSem[i], 0, 0);



235        CHECK_STATUS(status, 0, "sem_init: subrateTaskSem");



236        // g_sp.sched_priority = MW_BASERATE_PRIORITY -40 + subratePriority[i] - 1;



237        g_sp.sched_priority = subratePriority[i];



238#if MW_RTOS_DEBUG



239        PX4_INFO("MW_SUBRATE_PRIORITY = %d ", (int)g_sp.sched_priority);



240#endif



241



242#if defined(MW_PX4_NUTTX_BUILD)



243        status = pthread_attr_setschedparam(&attr, &g_sp);



244        CHECK_STATUS(status, 0, "pthread_attr_setschedparam");



245        status = pthread_create(&subRateThread[i], &attr, &subrateTask, (void*)&taskId[i]);



246#elif defined(MW_PX4_POSIX_BUILD)



247        status = pthread_create(&subRateThread[i], nullptr, &subrateTask, (void*)&taskId[i]);



248#endif



249        CHECK_STATUS(status, 0, "pthread_create");



250



251#if MW_RTOS_DEBUG



252        PX4_INFO("** Sub Rate Task ID = %d ", subRateThread[i]);



253        fflush(stdout);



254#endif



255    }



256#endif // End of "If Multiple Rates" check



257



258#if defined(MW_PX4_NUTTX_BUILD)



259    g_sp.sched_priority = baseRatePriority;



260    status = pthread_attr_setschedparam(&attr, &g_sp);



261#endif



262



263    /* Create the Scheduler Task here */



264    long stMicroSec;



265    stMicroSec = (long)(g_info.period * 1e6);



266    hrt_call_every(&BaseRate_HRT, stMicroSec, stMicroSec, Base_HRT_Callback,



267                   NULL); // timing callback to post base-rate semaphore



268



269/* Create the External Mode Task here */



270#ifdef EXT_MODE



271    /*---------- Creating EXT-mode Background Task ------------------*/



272    /* Setup signal info to block for EXT Mode */



273    EXT_sig_info.period = 0.1; // Sample rate at which we want Simulink to update w/ ext mode



274    EXT_sig_info.sigNo = SIGRTMIN;



275    /* Signaling the thread not working for low sample frequency*/



276    // sigemptyset(&EXT_sig_info.sigMask);



277    // sigaddset(&EXT_sig_info.sigMask, EXT_sig_info.sigNo);



278    // myAddBlockForThisEvent(EXT_sig_info.sigNo);



279



280    /* Setup thread for Custom Task */



281    fflush(stdout);



282#if defined(MW_PX4_NUTTX_BUILD)



283    g_sp.sched_priority = 40; // default is 100



284    status = pthread_attr_setschedparam(&attr, &g_sp);



285    status = pthread_create(&backgroundThread, &attr, &backgroundTask, (void*)&EXT_sig_info);



286#elif defined(MW_PX4_POSIX_BUILD)



287    status = pthread_create(&backgroundThread, nullptr, &backgroundTask, (void*)&EXT_sig_info);



288#endif



289    CHECK_STATUS(status, 0, "EXT Mode Background pthread_create");



290/*---------------------------------------------------------*/



291#endif

296    fflush(stdout);



297#if defined(MW_PX4_NUTTX_BUILD)



298    g_sp.sched_priority = 40; // default is 100

301#elif defined(MW_PX4_POSIX_BUILD)



302    status = pthread_create(&loggingThread, nullptr, &loggingThreadFcn, NULL);



303#endif



304    CHECK_STATUS(status, 0, "SD Card Background pthread_create");



305#if MW_RTOS_DEBUG



306    PX4_INFO("** SD Card Background ID = %d ", loggingThread);



307    fflush(stdout);



308#endif



309



310/*---------------------------------------------------------*/



311#endif



312



313



314#if MW_RTOS_DEBUG



315    PX4_INFO("px4_simulink_app : Created Simulink task threads successfully\n");



316    fflush(stdout);



317#endif



318#if defined(MW_PX4_NUTTX_BUILD)



319    sched_unlock();



320#endif
